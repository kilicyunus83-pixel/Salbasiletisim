function printZRaporu() { const now = new Date(); const isToday = (dStr) => { const d = new Date(dStr); return d.getDate() === now.getDate() && d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear(); }; const salesToday = db.sales.filter(s => !s.isRefunded && isToday(s.date)); const finsToday = db.finances.filter(f => isToday(f.date)); const ciro = salesToday.reduce((sum, s) => sum + s.total, 0); // Toplam ciro (Veresiye dahil) const kar = salesToday.reduce((sum, s) => sum + (s.profit || 0), 0); // Satışlardan edilen kâr // Kasaya bugün giren net fiziki nakit (Peşin satışlar + Veresiye tahsilatları) const gelenNakit = finsToday.filter(f => f.type === 'Gelir').reduce((sum, f) => sum + f.amount, 0); // Kasadan bugün çıkan masraf/giderler const gider = finsToday.filter(f => f.type === 'Gider').reduce((sum, f) => sum + f.amount, 0); // Akşam dükkanı kapatırken çekmecede olması gereken net nakit para const netKasa = gelenNakit - gider; const settings = JSON.parse(localStorage.getItem('receiptSettings')) || { companyName: 'SALBAŞ İLETİŞİM' }; let html = `

${settings.companyName}

Z RAPORU (GÜN SONU)

${now.toLocaleString('tr-TR')}

`; html += `
Toplam Satış İşlemi:${salesToday.length} Adet
Toplam Ciro (Veresiye Dahil):₺${ciro.toFixed(2)}
Günlük Net Kâr:₺${kar.toFixed(2)}
Kasaya Toplam Giriş:₺${gelenNakit.toFixed(2)}
Toplam Dükkan Gideri:₺${gider.toFixed(2)}
ÇEKMECEDEKİ NAKİT:₺${netKasa.toFixed(2)}
`; document.getElementById('print-area').innerHTML = html; document.getElementById('print-area').classList.add('active-print'); window.print(); setTimeout(() => document.getElementById('print-area').classList.remove('active-print'), 500); }
