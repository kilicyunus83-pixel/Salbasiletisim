/* ======================================================== MÜŞTERİ DÜZENLEME VE KAYDETME FONKSİYONLARI ======================================================== */ function editCustomer(i) { const c = db.customers[i]; if(!c) return; document.getElementById('m-c-edit-idx').value = i; document.getElementById('m-c-name').value = c.name; document.getElementById('m-c-tel').value = c.tel || ''; document.getElementById('m-c-note').value = c.note || ''; document.getElementById('m-c-debt').value = c.debt; openModal('cust-modal'); } function saveCustomer() { const idx = parseInt(document.getElementById('m-c-edit-idx').value); const name = document.getElementById('m-c-name').value.trim(); const tel = document.getElementById('m-c-tel').value.trim(); const note = document.getElementById('m-c-note').value.trim(); const debt = Number(document.getElementById('m-c-debt').value) || 0; if(!name) return showToast('Müşteri adı boş bırakılamaz!', 'error'); const customerData = { name, tel, note, debt }; if(idx > -1) { db.customers[idx] = customerData; showToast('Müşteri bilgileri güncellendi.', 'success'); } else { db.customers.push(customerData); showToast('Yeni müşteri kaydedildi.', 'success'); } // Formu sıfırla document.getElementById('m-c-edit-idx').value = "-1"; document.getElementById('m-c-name').value = ""; document.getElementById('m-c-tel').value = ""; document.getElementById('m-c-note').value = ""; document.getElementById('m-c-debt').value = ""; sync(); closeModal(); }
